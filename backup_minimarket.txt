create database minimarket;
use minimarket;

create table Localidades(IdLocalidad int not null, nom_localidad varchar(60) not null, cod_post smallint not null, constraint PK_LOCALIDAD primary key (IdLocalidad));
create table Clientes(CUIT BIGINT not null, Apellido varchar(35) not null, Nombre varchar(30) not null, email varchar(40) not null, telefono varchar(20) not null, domicilio varchar(80) not null, IdLocalidad int not null, constraint PK_CUIT primary key (CUIT), constraint FK_LOCALIDAD foreign key (IdLocalidad) references Localidades (IdLocalidad));
create table Vendedores(CUIL bigint not null, Apellido varchar(35) not null, Nombre varchar(30) not null, email varchar(40) not null, telefono varchar(20) not null, domicilio varchar(80) not null, constraint PK_CUIL primary key (CUIL));
create table Facturas(IdFactura int not null auto_increment, num_factura int not null, fecha date not null, total float not null, CUIT bigint not null, CUIL bigint not null, constraint FK_CUIT foreign key (CUIT) references Clientes (CUIT), constraint FK_CUIL foreign key (CUIL) references Vendedores (CUIL), constraint PK_FACTURA primary key (IdFactura));
create table Proveedores(CUIT bigint not null, denominacion varchar(80) not null, email varchar(40) not null, whatsapp varchar(20) not null, constraint PK_CUIT primary key (CUIT));
create table Articulos(IdArticulo smallint not null, nom_articulo varchar(60) not null, constraint PK_ARTICULO primary key (IdArticulo));
create table ComprasArticulos(IdArticulo smallint not null, CUIT bigint not null, fecha_operacion date not null, cantidad int not null, precio_costo float not null, constraint PK_ARTICULO primary key (IdArticulo), constraint FK_CUIT_PROVE foreign key (CUIT) references Proveedores (CUIT));
create table DetalleFacturas(IdFactura int not null, IdArticulo smallint not null, cantidad int not null, precio_venta float not null, constraint PK_VENTA primary key (IdFactura, IdArticulo), constraint FK_FACTURA foreign key (IdFactura) references Facturas (IdFactura), constraint FK_ARTICULO foreign key (IdArticulo) references ComprasArticulos (IdArticulo));
